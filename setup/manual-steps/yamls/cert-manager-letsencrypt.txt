apiVersion: certmanager.k8s.io/v1alpha1
kind: ClusterIssuer
metadata:
  name: letsencrypt-prod
spec:
  acme:
    server: https://acme-v02.api.letsencrypt.org/directory
    email: $EMAIL
    privateKeySecretRef:
      name: letsencrypt-prod
    dns01:
      providers:
        - name: r53-dns
          route53:
            accessKeyID: $R53_ACCESS_KEY_ID
            region: us-central-1
            hostedZoneID: $R53_HOSTED_ZONE_ID
            apiKeySecretRef:
              name: valid-route53-creds
              key: $R53_SECRET_KEY
---
apiVersion: certmanager.k8s.io/v1alpha1
kind: Certificate
metadata:
  name: acme-crt
  namespace: kube-system
spec:
  secretName: acme-crt-secret
  commonName: "*.$DOMAIN"
  dnsNames:
  - $DOMAIN
  - '*.$DOMAIN'
  - '*.k.$DOMAIN'
  acme:
    config:
    - dns01:
        provider: r53-dns
      domains:
      - $DOMAIN
      - '*.$DOMAIN'
      - '*.k.$DOMAIN'
  issuerRef:
    name: letsencrypt-prod
    kind: ClusterIssuer
